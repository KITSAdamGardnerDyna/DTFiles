#!/bin/bash
sudo useradd dynatrace
sudo groupadd dynatrace
sudo usermod -a -G dynatrace dynatrace
sudo apt-get update -y
sudo apt-get install default-jre wget -y
sudo wget https://files.dynatrace.com/downloads/OnPrem/dynaTrace/7.0/7.0.0.2429/dynatrace-full-7.0.0.2429-linux-x86-64.jar -P /opt
sudo java -jar /opt/dynatrace-full-7.0.0.2429-linux-x86-64.jar -b 64 -t /opt/dynatrace
sudo chown -R dynatrace:dynatrace /opt/dynatrace
sudo cp /opt/dynatrace/init.d/dynaTraceServer /etc/init.d
sudo sed -i -e 's,DT_OPTARGS=,DT_OPTARGS="-sizing demo -memory demo",g' /etc/init.d/dynaTraceServer
sudo sed -i -e 's,DT_FE_OPTARGS=,DT_FE_OPTARGS="-sizing demo -memory demo",g' /etc/init.d/dynaTraceServer
sudo sed -i -e 's,DT_RUNASUSER=,DT_RUNASUSER=dynatrace,g' /etc/init.d/dynaTraceServer
sudo service dynaTraceServer start
